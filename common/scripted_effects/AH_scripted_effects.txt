check_area_dev = { #计算区域发展度、平均发展度。
	#0.初始化变量
	set_variable = {
		which = Num_$squence$
		value = 0
	}
	set_variable = {
		which = Num_total_$squence$
		value = 0
	}
	set_variable = {
		which = operator
		value = 0
	}
	set_variable = {
		which = dev_$squence$
		value = 0 
	}
	set_variable = {
		which = dev_tax_$squence$
		value = 0 
	}
	set_variable = {
		which = dev_production_$squence$
		value = 0 
	}
	set_variable = {
		which = dev_manpower_$squence$
		value = 0 
	}
	#1.导出省份发展度
	$fuya$ = { #省份数量，税收之和，生产之和，人力之和
		if = {
			limit = { owned_by = ROOT }
			owner = {
				change_variable = {
					which = Num_$squence$
					value = 1
				}
				change_variable = {
					which = Num_total_$squence$
					value = 1
				}
				export_to_variable = {
					which = operator
					value = base_tax
					who = PREV
				}
				change_variable = {
					which = dev_tax_$squence$
					which = operator
				}
				export_to_variable = {
					which = operator
					value = base_production
					who = PREV
				}
				change_variable = {
					which = dev_production_$squence$
					which = operator
				}
				export_to_variable = {
					which = operator
					value = base_manpower
					who = PREV
				}
				change_variable = {
					which = dev_manpower_$squence$
					which = operator
				}
			}
			
		}
		else = {
			ROOT = {
				change_variable = {
					which = Num_total_$squence$
					value = 1
				}
			}
		}
	}
	#2.税收、生产、人力求和，得到总发展度
	change_variable = { #税收
		which = dev_$squence$
		which = dev_tax_$squence$
	}
	change_variable = { #生产
		which = dev_$squence$
		which = dev_production_$squence$
	}
	change_variable = { #人力
		which = dev_$squence$
		which = dev_manpower_$squence$
	}
	#3.求平均发展度
	set_variable = {
		which = mean_dev_$squence$
		which = dev_$squence$
	}
	divide_variable = { 
		which = mean_dev_$squence$
		which = Num_$squence$
	}

	set_variable = {
		which = mean_dev_tax_$squence$
		which = dev_tax_$squence$
	}
	divide_variable = { 
		which = mean_dev_tax_$squence$
		which = Num_$squence$
	}

	set_variable = {
		which = mean_dev_production_$squence$
		which = dev_production_$squence$
	}
	divide_variable = { 
		which = mean_dev_production_$squence$
		which = Num_$squence$
	}

	set_variable = {
		which = mean_dev_manpower_$squence$
		which = dev_manpower_$squence$
	}
	divide_variable = { 
		which = mean_dev_manpower_$squence$
		which = Num_$squence$
	}
	#4.结束
	#得到了10个变量
	
	#check_area_dev使用说明
		# $fuya$ #府衙,使用时填入区域名称
		# $squence$ #序号，使用时填入的内容成为变量名称的一部分，所以要符合变量命名的规范。建议按顺序填数字。
		# 创建的变量如下
		# Num_$squence$ ：$fuya$区域全部的省份数量
		# Num_total_$squence$ ：在$fuya$区域拥有的省份数量
		# dev_tax_$squence$ ：区域已拥有省份的基础税收之和
		# dev_production_$squence$ ：区域已拥有省份的基础生产之和
		# dev_manpower_$squence$ ：区域已拥有省份的基础人力之和
		# dev_$squence$ ：区域已拥有省份的发展度之和
		# mean_dev_tax_$squence$ ：区域已拥有省份的平均基础税收
		# mean_dev_production_$squence$ ：区域已拥有省份的平均基础生产
		# mean_dev_manpower_$squence$ ：区域已拥有省份的平均基础人力
		# mean_dev_$squence$ ：区域已拥有省份的平均发展度	
}

	}

dev_bonus_remove_buff = {
	# buffname buff名称
	# mode 选择模式
	set_variable = { which = dev_bonus_remove_buff_mode value = $mode$ }
	if = { # mode == 0 ,彻底清除
		limit = { is_variable_equal = { which = dev_bonus_remove_buff_mode value = 0 } }
		if = {
			limit = { has_country_modifier = $buffname$_III }
			remove_country_modifier = $buffname$_III
		}
		if = {
			limit = { has_country_modifier = $buffname$_II }
			remove_country_modifier = $buffname$_II
		}
		if = {
			limit = { has_country_modifier = $buffname$_I }
			remove_country_modifier = $buffname$_I
		}
	}
	if = { # mode == 1 ,降级。也许以后会用的上吧
		limit = { is_variable_equal = { which = dev_bonus_remove_buff_mode value = 1 } }
		if = {
			limit = { has_country_modifier = $buffname$_III }
			remove_country_modifier = $buffname$_III
			add_country_modifier = {
				name = $buffname$_II
				duration = -1
			}
		}
		else_if = {
			limit = { has_country_modifier = $buffname$_II }
			remove_country_modifier = $buffname$_II
			add_country_modifier = {
				name = $buffname$_I
				duration = -1
			}
		}
		else_if = {
			limit = { has_country_modifier = $buffname$_I }
			remove_country_modifier = $buffname$_I
		}
	}
}
dev_bonus_change_upgrade_points_with_rule_4 = {
	# $chapter$
	# $page$
	if = {
		limit = {
			is_variable_equal = { which = b1c$chapter$p$page$_status value = 5 }
		}
		change_variable = { which = b1c$chapter$_upgrade_points value = 2 }
	}
	else_if = {
		limit = {
			check_variable = { which = b1c$chapter$p$page$_status value = 1 }
		}
		change_variable = { which = b1c$chapter$_upgrade_points value = 1 }
	}
	custom_tooltip = dev_bonus_change_upgrade_points_rule_4_tooltip
}

dev_bonus_change_upgrade_points_with_rule_6 = {
	# $chapter$
	# $page$
	if = {
		limit = {
			OR = {
				is_variable_equal = { which = b1c$chapter$p$page$_status value = 5 }
				is_variable_equal = { which = b1c$chapter$p$page$_status value = 4 }
			}
		}
		change_variable = { which = b1c$chapter$_upgrade_points value = 3 }
	}
	else_if = {
		limit = {
			check_variable = { which = b1c$chapter$p$page$_status value = 1 }
		}
		change_variable = { which = b1c$chapter$_upgrade_points value = 2 }
	}
	custom_tooltip = dev_bonus_change_upgrade_points_rule_6_tooltip
}
